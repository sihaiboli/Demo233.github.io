* 引入js

<pre>
    <code class="javascript">
        <script src="https://cdn.bootcss.com/jquery/1.12.4/jquery.min.js"></script>
    </code>
</pre>

* 编写body

{% highlight html linenos %}
<body>
	<div id="main">
		<a href="javascript:showBg();">点击这里查看效果</a> 
		<div id="fullbg">
	</div> 
	<div id="dialog"> 
		<p class="close"><a href="#" onclick="closeBg();">关闭</a></p> 
		<div>正在加载，请稍后....</div> 
	</div> 

</body>
{% endhighlight %}

* JQuery

<pre>
    <code class="javascript">
       
       function showBg() { 
       		var bh = $("body").height(); 
       		var bw = $("body").width(); 
       		$("#fullbg").css({ 
       			height:bh, 
       			width:bw, 
       			display:"block" 
       		}); 
       		$("#dialog").show(); 
       	} 
       	
       	//关闭灰色 jQuery 遮罩 
       	function closeBg() { 
       		$("#fullbg,#dialog").hide(); 
       	} 
       
    </code>
</pre>
